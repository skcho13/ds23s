def cor_heatmap(df, font=10, abs=False):
    if abs:
        df = np.abs(df)
    mask = np.triu(np.ones_like(df, dtype=bool))
    np.fill_diagonal(mask, False)
    cmap = sns.diverging_palette(205, 12, s=100, l=55, as_cmap=True)
    
    ax = sns.heatmap(df, annot=True, cmap=cmap, vmin=-1, vmax=1, mask=mask,
    annot_kws={"size": font})
    ax.set(xlabel="", ylabel="")
    ax.set_xticks(ax.get_xticks(), ax.get_xticklabels(), rotation=45, ha='right')

# parsing function
def make_datetime_100(data, time):
    date_data = (data[['year', 'month', 'day', time]].
        assign(
            hour = lambda x: x[time] // 100,
            minute = lambda x: x[time] % 100).
        filter(['year', 'month', 'day', 'hour', 'minute'])
    )
    
    out = pd.to_datetime(date_data)

    return out
